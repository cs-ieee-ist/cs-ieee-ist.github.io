    int  color;

    struct  node  *next;

}  *link;

void  initializeNodes(link*  list,  int  numNodes)  {

    int  i;

    for  (i  =  0;  i  <   numNodes;  i++)  {

        list[i]  =  malloc(sizeof(struct  node));

        list[i]-> value  =  i;

        list[i]-> next  =  NULL;

    }

}

}
